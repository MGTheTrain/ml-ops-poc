   kubectl get tfjobs -n {{ .Release.Namespace }}

   kubectl logs -f -n {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "keras-mnist-training.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -c tensorflow

   kubectl delete tfjob {{ include "keras-mnist-training.fullname" . }} -n {{ .Release.Namespace }}
